CREATE DATABASE totemsystem;

use totemsystem;

drop table Usuario;
CREATE TABLE IF NOT EXISTS Usuario(
id_usuario INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(255),
cpf varchar(45),
cargo varchar(45),
email VARCHAR(255),
senha VARCHAR(225),
telefone varchar(15)
);


CREATE TABLE IF NOT EXISTS tb_empresa(
id_empresa INT PRIMARY KEY AUTO_INCREMENT,
cnpj VARCHAR(255),
razao_social varchar(100),
rua_empresa varchar(100),
numero_empresa int,
complemento_empresa varchar(100),
telefone_empresa varchar(100),
email_empresa VARCHAR(255),
senha_empresa VARCHAR(255)
);

CREATE TABLE IF NOT EXISTS tb_totem(
id_totem INT PRIMARY KEY AUTO_INCREMENT,
data_atual varchar(45),
sistema_operacional varchar(45),
hostname varchar(45),
fabricante_sistema varchar(45),
arquitetura int,
inicializado_em varchar(45),
permissoes varchar(45),
marca varchar(45),
fabricante_processador varchar(45),
micro_arquitetura varchar(45),
cpus_fisicas int,
cpus_logicas int,
pacotes_fisicos int,
frequencia int);

CREATE TABLE IF NOT EXISTS tb_log(
id_log INT PRIMARY KEY AUTO_INCREMENT,
data_atual varchar(45),
memoria_em_uso BIGINT,
memoria_disponivel BIGINT,
total_processos INT,
total_threads INT,
total_de_servicos INT);
