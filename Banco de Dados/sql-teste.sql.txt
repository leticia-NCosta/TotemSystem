CREATE DATABASE totemsystem;

use totemsystem;

CREATE TABLE IF NOT EXISTS Usuario(
id INT PRIMARY KEY AUTO_INCREMENT,
nome VARCHAR(255),
email VARCHAR(255),
senha VARCHAR(15)
);



CREATE TABLE IF NOT EXISTS tb_empresa(
id_empresa INT PRIMARY KEY AUTO_INCREMENT,
cnpj VARCHAR(255),
razao_social varchar(100),
rua_empresa varchar(100),
numero_empresa int,
complemento_empresa varchar(100),
telefone_empresa varchar(100),
email_empresa VARCHAR(255),
senha_empresa VARCHAR(255)
);


CREATE TABLE IF NOT EXISTS tb_totem(
idtotem INT PRIMARY KEY AUTO_INCREMENT,
dataAtual varchar(45),
sistemaOperacional varchar(45),
hostname varchar(45),
fabricanteSistema varchar(45),
arquitetura int,
inicializadoEm varchar(45),
permissoes varchar(45),
marca varchar(45),
fabricanteProcessador varchar(45),
microArquitetura varchar(45),
cpusFisicas int,
cpusLogicas int,
pacotesFisicos int,
frequencia int);


CREATE TABLE IF NOT EXISTS tb_log(
idLog INT PRIMARY KEY AUTO_INCREMENT,
memoriaEmUso BIGINT,
memoriaDisponivel BIGINT,
totalProcessos INT,
totalThreads INT,
totalDeServicos INT);
